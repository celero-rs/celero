Celero — Rust 
#[api(get, path = "/users/{id}")]
async fn get_user(
    Path(id): Path<i64>,
    Db(db): Db,
) -> Result<Json<User>> {
    let user = UserEntity::find_by_id(id)
        .one(&db).await?
        .ok_or(Error::not_found())?;
    Ok(Json(user))
}
FastAPI — Python 
@app.get("/users/{id}")
async def get_user(
    id: int,
    db: Session = Depends(get_db),
) -> User:
    user = db.query(User).get(id)
    if not user:
        raise HTTPException(404)
    return user
